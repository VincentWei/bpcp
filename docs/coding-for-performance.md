### 版本一：按位对比

```c
int count_one_bits(unsigned char byte)
{
    int n = 0;

    for (int i = 0; i < 8; i++) {
        if ((byte >> i) & 0x01) {
            n++;
        }
    }

    return n;
}
```

static unsigned char nr_one_bits_half_byte[] = {
    0, 1, 1, 2, 1, 2, 2, 3,

static inline int count_one_bits(unsigned char byte)
{
    return nr_one_bits_half_byte[byte & 0x0F] +

### 动态缓冲区分配

```c
void foo(size_t len)
{
    char *buff = malloc(len);

    ...

    free(buff);
}
```

减少 `malloc` 的调用，优先使用栈缓冲区：

```c
void foo(size_t len)
{
    char stack_buff[1024];
    char *buff;

    if (len > 1024)
        buff = malloc(len);
    else
        buff = stack_buff;

    if (buff) {
        ...

        if (buff != stack_buff)
            free(buff);
    }
}
```
